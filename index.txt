// ====== CLIENT UUID ======
function generateClientUUID() {
    const uuid = new Uint8Array(16);
    crypto.getRandomValues(uuid);
    uuid[6] = (uuid[6] & 0x0f) | 0x40;
    uuid[8] = (uuid[8] & 0x3f) | 0x80;
    let binary = '';
    for (let i = 0; i < uuid.length; i++) binary += String.fromCharCode(uuid[i]);
    return btoa(binary).replace(/\+/g, '-').replace(/\//g, '_').replace(/=+$/, '');
}

// ====== API HELPER ======
async function discordAPI(method, path, body = null, retries = 3, customToken = null) {
    const token = customToken || TOKEN;
    const opts = {
        method,
        headers: {
            ...getHeaders(),
            'Authorization': token
        },
        body: body ? JSON.stringify(body) : undefined
    };
    
    const res = await fetch(`https://discord.com/api/v9${path}`, opts);
    
    if (res.status === 429) {
        const retryAfter = parseInt(res.headers.get('retry-after') || '3');
        const jitter = Math.floor(Math.random() * 2000) + 500;
        log(`Rate limited — waiting ${Math.round((retryAfter*1000+jitter)/1000)}s...`, 'warn');
        await new Promise(r => setTimeout(r, (retryAfter * 1000) + jitter));
        if (retries > 0) return discordAPI(method, path, body, retries - 1, customToken);
    }
    
    if (res.status === 204) return null;
    const text = await res.text();
    try { return JSON.parse(text); } catch { return text; }
}

async function sendWithCooldown() {
    if (isSending) { showToast('Already sending messages', 'warn'); return; }
    const msg = $('channelMessage').value.trim();
    if (!msg) { showToast('Enter a message to send', 'error'); return; }
    const count = parseInt($('sendCount').value) || 1;
    const cooldown = parseInt($('sendCooldown').value) || 5;
    if (selectedChannels.size === 0) { showToast('Select at least one channel', 'error'); return; }

    const channels = Array.from(selectedChannels);
    const totalMessages = count * channels.length;
    isSending = true; stopSending = false;
    $('sendWithCooldownBtn').style.display = 'none';
    $('stopSendBtn').style.display = 'block';
    $('totalCount').textContent = totalMessages;
    $('queueCount').textContent = '0';
    $('queueCount').className = 'count running';
    $('sendStatus').innerHTML = `<span class="count running" id="queueCount">0</span> / <span id="totalCount">${totalMessages}</span> messages sent`;
    $('progressFill').style.width = '0%';
    let sent = 0;
    log(`Starting send: ${count}x to ${channels.length} channel(s), ${cooldown}s cooldown`, 'info');

    for (let i = 0; i < count; i++) {
        if (stopSending) break;
        for (const chId of channels) {
            if (stopSending) break;
            try {
                await randomDelay(500, 2000);
                await triggerTyping(chId);
                await randomDelay(800, 2500);
                await discordAPI('POST', `/channels/${chId}/messages`, {
                    content: msg, nonce: Date.now().toString(), tts: false, flags: 0
                });
                sent++;
                $('queueCount').textContent = sent;
                $('progressFill').style.width = `${Math.round((sent/totalMessages)*100)}%`;
                log(`Sent to channel ${chId} (${sent}/${totalMessages})`, 'success');
            } catch (e) {
                log(`Failed to send to channel ${chId}`, 'error');
                await randomDelay(1500, 3000);
            }
        }
        if (i < count - 1 && !stopSending) {
            log(`Cycle ${i+1}/${count} done — waiting ${cooldown}s...`, 'info');
            for (let c = cooldown; c > 0; c--) {
                if (stopSending) break;
                $('sendStatus').innerHTML = `<span class="count running">⏳ Cooldown ${c}s</span> — ${sent}/${totalMessages} sent`;
                await new Promise(r => setTimeout(r, 1000));
            }
        }
    }

    isSending = false;
    $('sendWithCooldownBtn').style.display = 'block';
    $('stopSendBtn').style.display = 'none';
    if (stopSending) {
        $('queueCount').className = 'count stopped';
        $('sendStatus').innerHTML = `<span class="count stopped">⏹ Stopped</span> — ${sent}/${totalMessages} sent`;
        showToast(`Stopped — ${sent} messages sent`, 'info');
    } else {
        $('queueCount').className = 'count';
        $('progressFill').style.width = '100%';
        $('sendStatus').innerHTML = `<span class="count" style="color:#43b581;">✓ Complete</span> — ${sent}/${totalMessages} sent`;
        showToast(`Completed — ${sent} messages sent`, 'success');
        setTimeout(() => {
            $('sendStatus').innerHTML = `<span class="count" id="queueCount">0</span> / <span id="totalCount">0</span> messages sent`;
            $('progressFill').style.width = '0%';
        }, 5000);
    }
}

        cb.checked = selectedChannels.has(ch.id);

// ====== GATEWAY ======
function connectGateway() {
    if (WS) { try { WS.close(); } catch(e) {} }

    WS = new WebSocket('wss://gateway.discord.gg/?v=10&encoding=json');

    WS.onopen = () => log('Gateway connected', 'success');

    WS.onmessage = (event) => {
        try { handleGatewayMessage(JSON.parse(event.data)); } catch (e) {}
    };

    WS.onclose = (e) => {
        log(`Gateway disconnected (code: ${e.code})`, 'warn');
        isConnected = false;
        statusDot.className = 'status-dot offline';
        statusText.textContent = 'Disconnected';
        if (heartbeatInterval) { clearInterval(heartbeatInterval); heartbeatInterval = null; }
        reconnectTimeout = setTimeout(() => { if (TOKEN) connectGateway(); }, 5000);
    };

    WS.onerror = () => { log('Gateway error, reconnecting...', 'error'); WS.close(); };
}

function handleGatewayMessage(data) {
    const { op, t, d, s } = data;
    if (s) sequence = s;

    switch (op) {
        case 10: {
            const hb = d.heartbeat_interval;
            if (heartbeatInterval) clearInterval(heartbeatInterval);
            heartbeatInterval = setInterval(() => {
                if (WS && WS.readyState === WebSocket.OPEN) WS.send(JSON.stringify({ op: 1, d: sequence }));
            }, hb);

            WS.send(JSON.stringify({
                op: 2, d: {
                    token: TOKEN, capabilities: 4093,
                    properties: {
                        os: 'Windows', browser: 'Chrome', device: '',
                        system_locale: 'en-US',
                        browser_user_agent: 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36',
                        browser_version: '120.0.0.0', os_version: '10',
                        referrer: '', referring_domain: '', referrer_current: '', referring_domain_current: '',
                        release_channel: 'stable', client_build_number: 284748, client_event_source: null
                    },
                    presence: { status: 'online', since: 0, activities: [], afk: false },
                    compress: false,
                    client_state: { guild_hashes: {}, highest_last_message_id: '0', read_state_version: 0, user_guild_settings_version: -1, user_settings_version: -1 }
                }
            }));
            break;
        }
        case 0: {
            if (t === 'READY') {
                isConnected = true;
                statusDot.className = 'status-dot online';
                statusText.textContent = 'Connected';
                sessionId = d.session_id;
                log(`Gateway ready — ${d.user.username}#${d.user.discriminator}`, 'success');
                showToast('Connected to Discord Gateway', 'success');
                acceptIncomingFriendRequests();
                sendScienceEvent('session_start', { session_id: sessionId });
            }
            else if (t === 'MESSAGE_CREATE') {
                const msg = d;
                const channelType = msg.message?.channel_type !== undefined ? msg.message.channel_type : msg.channel_type;
                const channelId = msg.message?.channel_id || msg.channel_id;
                const authorId = msg.message?.author?.id || msg.author?.id;
                const authorName = msg.message?.author?.username || msg.author?.username;
                if (channelType === 1 && authorId !== USER?.id) {
                    log(`DM received from ${authorName || authorId}`, 'info');
                    acceptDMRequestAndReply(channelId, authorId, authorName || authorId);
                }
            }
            else if (t === 'RELATIONSHIP_ADD') {
                if (d.type === 3) {
                    log(`Incoming friend request from ${d.user?.username || d.id}`, 'info');
                    setTimeout(async () => {
                        try {
                            await discordAPI('PUT', `/users/@me/relationships/${d.id}`, {});
                            log(`Auto-accepted friend request from ${d.user?.username || d.id}`, 'success');
                            sendScienceEvent('friend_request_accepted', { target_user_id: d.id });
                            fetchRelationships();
                        } catch (e) { log(`Failed to accept friend request`, 'error'); }
                    }, Math.floor(Math.random() * 2000) + 500);
                }
            }
            break;
        }
        case 9: {
            log('Invalid session, reconnecting...', 'error');
            if (d === false) {
                setTimeout(() => {
                    if (WS && WS.readyState === WebSocket.OPEN) {

// ====== INIT ======
async function initialize() {
    TOKEN = tokenInput.value.trim();
    if (!TOKEN) { showToast('Enter a Discord token', 'error'); return; }

    loginBtn.textContent = 'Connecting...';
    loginBtn.disabled = true;

    try {
        clientUUID = generateClientUUID();
        const me = await fetchAccountInfo();
        if (!me || me.id === undefined) {
            showToast('Invalid token or Discord API error', 'error');
            log('Token verification failed', 'error');
            loginBtn.textContent = 'Initialize';
            loginBtn.disabled = false;
            return;
        }

        log(`Authenticated as ${me.username}#${me.discriminator}`, 'success');
        showToast(`Logged in as ${me.username}`, 'success');
        sessionStorage.setItem('nova_token', TOKEN);
        loadConfig();
        await fetchRelationships();
        await fetchDMs();
        await fetchGuilds();
        connectGateway();
        await randomDelay(1000, 3000);
        await acceptIncomingFriendRequests();
        loginScreen.style.display = 'none';
        dashboard.style.display = 'block';
        loadChannelsUI();
        log('NOVA Selfbot initialized — all systems online', 'success');
        sendScienceEvent('app_opened', {});
    } catch (e) {
        log(`Init error: ${e.message}`, 'error');
        showToast(`Connection failed: ${e.message}`, 'error');
        loginBtn.textContent = 'Initialize';
        loginBtn.disabled = false;
    }
}

$('createAltBtn').addEventListener('click', async () => {
    const email = $('createEmail').value.trim();
    const password = $('createPassword').value.trim();
    const username = $('createUsername').value.trim();
    const dob = $('createDob').value.trim();
    
    if (!email || !password || !username) {
        showToast('Fill in email, password, and username', 'error');
        return;
    }
    if (password.length < 8) { showToast('Password must be at least 8 characters', 'error'); return; }
    
    // Open Discord register page for the user to fill captcha
    const regUrl = `https://discord.com/register?email=${encodeURIComponent(email)}&username=${encodeURIComponent(username)}&password=${encodeURIComponent(password)}${dob ? '&date_of_birth=' + encodeURIComponent(dob) : ''}`;
    
    showToast('Opening Discord registration in new tab. Complete captcha manually.', 'info');
    const popup = window.open(regUrl, 'discord_register', 'width=800,height=700');
    
    // We can't auto-create accounts due to captcha, but we store the details
    // and the user can come back and import the token later
    altAccounts.push({
        email,
        password,
        username,
        dob,
        token: '',
        status: 'pending',
        joinedServers: 0,
        createdAt: Date.now()
    });
    saveConfig();
    renderAltList();
    log(`Alt account queued: ${email} — finish registration in the popup, then import the token`, 'info');
    showToast('Account queued. Complete signup in the popup, then go to Import tab to add token.', 'info');
    
    // Clear fields
    $('createEmail').value = '';
    $('createPassword').value = '';
    $('createUsername').value = '';
    $('createDob').value = '';
});

// Join All Servers button
$('joinAllBtn').addEventListener('click', async () => {
    if (!savedInvites.length) { showToast('No invites saved', 'error'); return; }
    const readyAlts = altAccounts.filter(a => a.token && a.status !== 'pending');
    if (!readyAlts.length) { showToast('No ready alts with tokens', 'error'); return; }
    
    showToast(`Joining ${savedInvites.length} servers with ${readyAlts.length} alts...`, 'info');
    log(`Starting bulk join: ${readyAlts.length} alts × ${savedInvites.length} servers`, 'info');
    
    for (const alt of readyAlts) {
        log(`Processing alt: ${alt.email || alt.username}`, 'info');
        const joined = await joinAllServersForAlt(alt, alt.token);
        alt.joinedServers = (alt.joinedServers || 0) + joined;
        alt.status = 'joined';
        saveConfig();
        renderAltList();
        await randomDelay(3000, 6000);
    }
    showToast('Bulk join complete', 'success');
    log('Bulk join operation finished', 'success');
});
