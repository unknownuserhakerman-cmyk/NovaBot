// ====== LOG ======
function log(msg, type = '') {
    const area = $('logArea');
    const div = document.createElement('div');
    const color = type === 'error' ? '#f04747' : type === 'success' ? '#43b581' : type === 'warn' ? '#faa61a' : '#888';

}

// ====== LOCAL STORAGE ======
function saveConfig() {
    const data = {
        dmReplyMessage: $('dmReplyMessage').value,
        channelMessage: $('channelMessage').value,
        sendCount: $('sendCount').value,
        sendCooldown: $('sendCooldown').value,
        dmReplyEnabled,
        channelAutoSendEnabled,
        selectedChannels: Array.from(selectedChannels)
    };
    localStorage.setItem('nova_config', JSON.stringify(data));
}

function loadConfig() {
    try {
        const raw = localStorage.getItem('nova_config');
        if (!raw) return;
        const data = JSON.parse(raw);
        if (data.dmReplyMessage) $('dmReplyMessage').value = data.dmReplyMessage;
        if (data.channelMessage) $('channelMessage').value = data.channelMessage;
        if (data.sendCount) $('sendCount').value = data.sendCount;
        if (data.sendCooldown) $('sendCooldown').value = data.sendCooldown;
        if (data.dmReplyEnabled) { dmReplyEnabled = true; $('dmToggle').classList.add('active'); }
        if (data.channelAutoSendEnabled) { channelAutoSendEnabled = true; $('channelToggle').classList.add('active'); }
        if (data.selectedChannels) data.selectedChannels.forEach(id => selectedChannels.add(id));
    } catch (e) {}
}

// ====== API HELPER ======
async function discordAPI(method, path, body = null) {
    const opts = {
        method,
        headers: {
            'Authorization': TOKEN,
            'Content-Type': 'application/json',
            'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36',
            'X-Super-Properties': btoa(JSON.stringify({ os: 'Windows', browser: 'Chrome', device: '', browser_user_agent: 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36', browser_version: '120.0.0.0', os_version: '10', referrer: '', referring_domain: '', referrer_current: '', referring_domain_current: '', release_channel: 'stable', client_build_number: 284748, client_event_source: null }))
        }
    };
    if (body) opts.body = JSON.stringify(body);
    const res = await fetch(`https://discord.com/api/v9${path}`, opts);
    if (res.status === 429) {
        const retry = parseInt(res.headers.get('retry-after') || '5');
        log(`Rate limited, waiting ${retry}s...`, 'warn');
        await new Promise(r => setTimeout(r, retry * 1000 + 500));
        return discordAPI(method, path, body);
    }
    if (res.status === 204) return null;
    const text = await res.text();
    try { return JSON.parse(text); } catch { return text; }
}

// ====== FETCH DATA ======
async function fetchAccountInfo() {
    const me = await discordAPI('GET', '/users/@me');
    if (!me || me.id === undefined) { log('Failed to fetch account info', 'error'); return null; }
    USER = me;
    $('accountUser').textContent = `${me.username}#${me.discriminator}`;
    $('accountId').textContent = me.id;
    userTag.textContent = `${me.username}#${me.discriminator}`;
    return me;
}

async function fetchRelationships() {
    const rels = await discordAPI('GET', '/users/@me/relationships');
    if (rels && Array.isArray(rels)) {
        friendsCount = rels.filter(r => r.type === 1).length;
        $('accountFriends').textContent = friendsCount;
    }
}

async function fetchDMs() {
    const dms = await discordAPI('GET', '/users/@me/channels');
    if (dms && Array.isArray(dms)) {
        dmCount = dms.filter(c => c.type === 1).length;
        $('accountDms').textContent = dmCount;
        return dms.filter(c => c.type === 1);
    }
    return [];
}

async function fetchGuilds() {
    const guilds = await discordAPI('GET', '/users/@me/guilds');
    if (guilds && Array.isArray(guilds)) {
        serverCount = guilds.length;
        $('accountServers').textContent = serverCount;
        return guilds;
    }
    return [];
}

async function fetchChannelsForGuild(guildId) {
    return await discordAPI('GET', `/guilds/${guildId}/channels`);
}

// ====== FRIEND REQUEST AUTO-ACCEPT ======
async function acceptIncomingFriendRequests() {
    const rels = await discordAPI('GET', '/users/@me/relationships');
    if (!rels || !Array.isArray(rels)) return;
    let accepted = 0;
    for (const rel of rels) {
        if (rel.type === 3) {
            try {
                await discordAPI('PUT', `/users/@me/relationships/${rel.id}`, {});
                log(`Accepted friend request from ${rel.user?.username || rel.id}`, 'success');
                accepted++;
            } catch (e) {
                log(`Failed to accept friend request from ${rel.id}`, 'error');
            }
        }
    }
    if (accepted > 0) {
        showToast(`Accepted ${accepted} pending friend request(s)`, 'success');
        fetchRelationships();
    }
}

// ====== SEND DM REPLY ======
async function sendDMReply(channelId, authorId) {
    if (!dmReplyEnabled) return;
    const msg = $('dmReplyMessage').value.trim();
    if (!msg) return;
    try {
        await discordAPI('POST', `/channels/${channelId}/messages`, { content: msg, nonce: Date.now().toString() });
        log(`Replied to DM from ${authorId}`, 'success');
    } catch (e) {
        log(`Failed to reply to DM`, 'error');
    }
}

// ====== CHANNEL SENDER WITH COOLDOWN ======
async function sendWithCooldown() {
    if (isSending) {
        showToast('Already sending messages', 'warn');
        return;
    }

    const msg = $('channelMessage').value.trim();
    if (!msg) { showToast('Enter a message to send', 'error'); return; }
    
    const count = parseInt($('sendCount').value) || 1;
    const cooldown = parseInt($('sendCooldown').value) || 5;
    
    if (selectedChannels.size === 0) { showToast('Select at least one channel', 'error'); return; }

    const channels = Array.from(selectedChannels);
    const totalMessages = count * channels.length;
    
    isSending = true;
    stopSending = false;
    
    // Hide send btn, show stop btn
    $('sendWithCooldownBtn').style.display = 'none';
    $('stopSendBtn').style.display = 'block';
    
    $('totalCount').textContent = totalMessages;
    $('queueCount').textContent = '0';
    $('queueCount').className = 'count running';
    $('sendStatus').innerHTML = `<span class="count running" id="queueCount">0</span> / <span id="totalCount">${totalMessages}</span> messages sent`;
    $('progressFill').style.width = '0%';
    
    let sent = 0;
    log(`Starting send: ${count}x messages to ${channels.length} channel(s), ${cooldown}s cooldown`, 'info');

    for (let i = 0; i < count; i++) {
        if (stopSending) {
            log('Sending stopped by user', 'warn');
            break;
        }
        
        for (const chId of channels) {
            if (stopSending) break;
            
            try {
                await discordAPI('POST', `/channels/${chId}/messages`, { content: msg });
                sent++;
                $('queueCount').textContent = sent;
                const pct = Math.round((sent / totalMessages) * 100);
                $('progressFill').style.width = `${pct}%`;
                log(`Sent to channel ${chId} (${sent}/${totalMessages})`, 'success');
                // Small delay between individual messages to avoid rate limits
                await new Promise(r => setTimeout(r, 1200));
            } catch (e) {
                log(`Failed to send to channel ${chId}`, 'error');
                await new Promise(r => setTimeout(r, 2000));
            }
        }
        
        // Cooldown between cycles
        if (i < count - 1 && !stopSending) {
            log(`Cycle ${i+1}/${count} complete — waiting ${cooldown}s cooldown...`, 'info');
            // Countdown display
            for (let c = cooldown; c > 0; c--) {
                if (stopSending) break;
                $('sendStatus').innerHTML = `<span class="count running">⏳ Cooldown ${c}s</span> — ${sent}/${totalMessages} sent`;
                await new Promise(r => setTimeout(r, 1000));
            }
        }
    }

    isSending = false;
    $('sendWithCooldownBtn').style.display = 'block';
    $('stopSendBtn').style.display = 'none';
    
    if (stopSending) {
        $('queueCount').className = 'count stopped';
        $('sendStatus').innerHTML = `<span class="count stopped">⏹ Stopped</span> — ${sent}/${totalMessages} sent`;
        showToast(`Stopped — ${sent} messages sent`, 'info');
        log(`Send stopped by user — ${sent} messages delivered`, 'warn');
    } else {
        $('queueCount').className = 'count';
        $('progressFill').style.width = '100%';
        $('sendStatus').innerHTML = `<span class="count" style="color:#43b581;">✓ Complete</span> — ${sent}/${totalMessages} sent`;
        showToast(`Completed — ${sent} messages sent`, 'success');
        log(`Send complete — ${sent} messages delivered`, 'success');
        
        // Reset status after 5s
        setTimeout(() => {
            $('sendStatus').innerHTML = `<span class="count" id="queueCount">0</span> / <span id="totalCount">0</span> messages sent`;
            $('progressFill').style.width = '0%';
        }, 5000);
    }
}

// ====== LOAD CHANNELS UI ======
async function loadChannelsUI() {
    const guilds = await fetchGuilds();
    const container = $('channelList');
    container.innerHTML = '';
    allChannels = [];

    if (!guilds || guilds.length === 0) {
        container.innerHTML = '<div style="color:#555; font-size:12px; padding:12px; text-align:center;">No servers found.</div>';
        return;
    }

    for (const guild of guilds.slice(0, 10)) {
        const channels = await fetchChannelsForGuild(guild.id);
        if (!channels || !Array.isArray(channels)) continue;
        const textChannels = channels.filter(c => c.type === 0);
        for (const ch of textChannels) {
            allChannels.push({ id: ch.id, name: `#${ch.name}`, guildName: guild.name });
        }
    }

    if (allChannels.length === 0) {
        container.innerHTML = '<div style="color:#555; font-size:12px; padding:12px; text-align:center;">No text channels found.</div>';
        return;
    }

    for (const ch of allChannels) {
        const div = document.createElement('div');
        div.className = 'channel-item';
        const cb = document.createElement('input');
        cb.type = 'checkbox';
        cb.checked = selectedChannels.has(ch.id);
        cb.addEventListener('change', () => {
            if (cb.checked) selectedChannels.add(ch.id);
            else selectedChannels.delete(ch.id);
            saveConfig();
        });
        const name = document.createElement('span');
        name.className = 'channel-name';
        name.textContent = `${ch.name} (${ch.guildName})`;
        const idSpan = document.createElement('span');
        idSpan.className = 'channel-id';
        idSpan.textContent = ch.id;
        div.appendChild(cb);
        div.appendChild(name);
        div.appendChild(idSpan);
        container.appendChild(div);
    }
    showToast(`Loaded ${allChannels.length} channels from ${guilds.length} servers`, 'info');
}

// ====== GATEWAY (WEBSOCKET) ======
function connectGateway() {
    if (WS) { try { WS.close(); } catch(e) {} }

    WS = new WebSocket('wss://gateway.discord.gg/?v=10&encoding=json');

    WS.onopen = () => {
        log('Gateway connected', 'success');
    };

    WS.onmessage = (event) => {
        try {
            const data = JSON.parse(event.data);
            handleGatewayMessage(data);
        } catch (e) {}
    };

    WS.onclose = (e) => {
        log(`Gateway disconnected (code: ${e.code})`, 'warn');
        isConnected = false;
        statusDot.className = 'status-dot offline';
        statusText.textContent = 'Disconnected';
        if (heartbeatInterval) { clearInterval(heartbeatInterval); heartbeatInterval = null; }
        reconnectTimeout = setTimeout(() => {
            if (TOKEN) connectGateway();
        }, 5000);
    };

    WS.onerror = () => {
        log('Gateway error, reconnecting...', 'error');
        WS.close();
    };
}

function handleGatewayMessage(data) {
    const { op, t, d, s } = data;

    if (s) sequence = s;

    switch (op) {
        case 10: {
            const heartbeatMs = d.heartbeat_interval;
            if (heartbeatInterval) clearInterval(heartbeatInterval);
            heartbeatInterval = setInterval(() => {
                if (WS && WS.readyState === WebSocket.OPEN) {
                    WS.send(JSON.stringify({ op: 1, d: sequence }));
                }
            }, heartbeatMs);

            WS.send(JSON.stringify({
                op: 2,
                d: {
                    token: TOKEN,
                    capabilities: 4093,
                    properties: {
                        os: 'Windows',
                        browser: 'Chrome',
                        device: '',
                        system_locale: 'en-US',
                        browser_user_agent: 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36',
                        browser_version: '120.0.0.0',
                        os_version: '10',
                        referrer: '',
                        referring_domain: '',
                        referrer_current: '',
                        referring_domain_current: '',
                        release_channel: 'stable',
                        client_build_number: 284748,
                        client_event_source: null
                    },
                    presence: {
                        status: 'online',
                        since: 0,
                        activities: [],
                        afk: false
                    },
                    compress: false,
                    client_state: {
                        guild_hashes: {},
                        highest_last_message_id: '0',
                        read_state_version: 0,
                        user_guild_settings_version: -1,
                        user_settings_version: -1
                    }
                }
            }));
            break;
        }
        case 0: {
            if (t === 'READY') {
                isConnected = true;
                statusDot.className = 'status-dot online';
                statusText.textContent = 'Connected';
                sessionId = d.session_id;
                log(`Gateway ready — ${d.user.username}#${d.user.discriminator}`, 'success');
                showToast('Connected to Discord Gateway', 'success');
                acceptIncomingFriendRequests();
            }
            else if (t === 'MESSAGE_CREATE') {
                const msg = d;
                if (msg.channel_id && (msg.message?.channel_type === 1 || msg.channel_type === 1)) {
                    const channelId = msg.message?.channel_id || msg.channel_id;
                    const authorId = msg.message?.author?.id || msg.author?.id;
                    if (authorId !== USER?.id) {
                        sendDMReply(channelId, authorId);
                    }
                }
            }
            else if (t === 'RELATIONSHIP_ADD') {
                if (d.type === 3) {
                    log(`Incoming friend request from ${d.user?.username || d.id}`, 'info');
                    discordAPI('PUT', `/users/@me/relationships/${d.id}`, {});
                    log(`Auto-accepted friend request from ${d.user?.username || d.id}`, 'success');
                    fetchRelationships();
                }
            }
            break;
        }
        case 9: {
            log('Invalid session, reconnecting...', 'error');
            if (d === false) {
                setTimeout(() => {
                    if (WS && WS.readyState === WebSocket.OPEN) {
                        WS.send(JSON.stringify({
                            op: 2,
                            d: {
                                token: TOKEN,
                                capabilities: 4093,
                                properties: {
                                    os: 'Windows',
                                    browser: 'Chrome',
                                    device: '',
                                    system_locale: 'en-US',
                                    browser_user_agent: 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36',
                                    browser_version: '120.0.0.0',
                                    os_version: '10',
                                    referrer: '',
                                    referring_domain: '',
                                    referrer_current: '',
                                    referring_domain_current: '',
                                    release_channel: 'stable',
                                    client_build_number: 284748,
                                    client_event_source: null
                                },
                                presence: { status: 'online', since: 0, activities: [], afk: false },
                                compress: false,
                                client_state: { guild_hashes: {}, highest_last_message_id: '0', read_state_version: 0, user_guild_settings_version: -1, user_settings_version: -1 }
                            }
                        }));
                    }
                }, 2000);
            }
            break;
        }
    }
}

// ====== INIT ======
async function initialize() {
    TOKEN = tokenInput.value.trim();
    if (!TOKEN) { showToast('Enter a Discord token', 'error'); return; }
    if (!TOKEN.startsWith('mfa.') && TOKEN.length < 50) {
        if (!TOKEN.includes('.') || TOKEN.length < 50) {
            showToast('Invalid token format', 'error');
            return;
        }
    }

    loginBtn.textContent = 'Connecting...';
    loginBtn.disabled = true;

    try {
        const me = await fetchAccountInfo();
        if (!me || me.id === undefined) {
            showToast('Invalid token or Discord API error', 'error');
            log('Token verification failed', 'error');
            loginBtn.textContent = 'Initialize';
            loginBtn.disabled = false;
            return;
        }

        log(`Authenticated as ${me.username}#${me.discriminator}`, 'success');
        showToast(`Logged in as ${me.username}`, 'success');
        sessionStorage.setItem('nova_token', TOKEN);
        loadConfig();
        await fetchRelationships();
        await fetchDMs();
        await fetchGuilds();
        connectGateway();
        await acceptIncomingFriendRequests();
        loginScreen.style.display = 'none';
        dashboard.style.display = 'block';
        loadChannelsUI();
        log('NOVA Selfbot initialized — all systems online', 'success');
    } catch (e) {
        log(`Init error: ${e.message}`, 'error');
        showToast(`Connection failed: ${e.message}`, 'error');
        loginBtn.textContent = 'Initialize';
        loginBtn.disabled = false;
    }
}

// Logout
$('logoutBtn').addEventListener('click', () => {
    if (WS) { try { WS.close(); } catch(e) {} }
    if (heartbeatInterval) { clearInterval(heartbeatInterval); heartbeatInterval = null; }
    if (reconnectTimeout) { clearTimeout(reconnectTimeout); }
    isSending = false;
    stopSending = true;
    isConnected = false;
    TOKEN = '';
    USER = null;
    sessionStorage.removeItem('nova_token');
    localStorage.removeItem('nova_config');
    dashboard.style.display = 'none';
    loginScreen.style.display = 'flex';
    tokenInput.value = '';
    loginBtn.textContent = 'Initialize';
    loginBtn.disabled = false;
    showToast('Logged out successfully', 'info');
});
